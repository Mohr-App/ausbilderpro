Firestore-Regeln prüfen
Damit Ausbilder, Lehrgangsarten und Berufsgruppen gespeichert werden können, muss shared/kursSettings in den Firestore Security Rules erlaubt sein.
Firebase Console → Firestore → Regeln — ersetze die bestehenden Regeln durch: 
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isAdmin() {
      return request.auth != null
        && (get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin'
         || get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'verwaltung');
    }

    match /users/{uid} {
      allow read, write: if request.auth != null && request.auth.uid == uid;
      allow read, write: if isAdmin();
      match /{sub}/{subDoc} {
        allow read, write: if request.auth != null && request.auth.uid == uid;
        allow read, write: if isAdmin();
        match /{deep}/{deepDoc} {
          allow read, write: if request.auth != null && request.auth.uid == uid;
          allow read, write: if isAdmin();
        }
      }
    }

    match /shared/{docId} {
      allow read: if request.auth != null;
      allow write: if isAdmin();
      match /{sub}/{subDoc} {
        allow read: if request.auth != null;
        allow write: if isAdmin();
        // Ausbilder darf eigenen Urlaub aktualisieren
        allow update: if request.auth != null
          && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['urlaub']);
        // Ausbilder/Admin darf Benachrichtigungen aktualisieren
        allow update: if request.auth != null
          && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['benachrichtigungen']);
        // Ausbilder darf Berufsschultage auf Kurs setzen
        allow update: if request.auth != null
          && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['berufsschultage']);
        // Urlaubslog: jeder darf Einträge erstellen
        allow create: if request.auth != null && docId == 'urlaubLog';
        match /{deep}/{deepDoc} {
          allow read: if request.auth != null;
          allow write: if isAdmin();
          // Ausbilder darf TN-Status + Attendance-Aggregat aktualisieren
          allow update: if request.auth != null
            && request.resource.data.diff(resource.data).affectedKeys().hasOnly(
              ['status','attTotal','attPresent','attAbsentE','attAbsentU','attSick','attBerufsschule','attLate','attEarly','attPercent','attLastSync','noten_mitarbeit','noten_fuehrung','noten_praxis','noten_extra']
            );
          // Azubi darf Kursbestätigung schreiben
          allow update: if request.auth != null
            && request.resource.data.diff(resource.data).affectedKeys().hasOnly(
              ['bestaetigtAm','bestaetigtVon']
            );
          // Ausbilder darf Dokumente hochladen/löschen (dokumente-Subcollection)
          allow create, delete: if request.auth != null && deep == 'dokumente';
        }
      }
    }

    match /globalTN/{doc} {
      allow read: if request.auth != null;
      allow write: if isAdmin();
      // Ausbilder darf nur status + wiederholer Felder ändern
      allow update: if request.auth != null
        && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['status', 'wiederholer']);
      // Azubi darf eigenes Profilbild setzen
      allow update: if request.auth != null
        && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['photoUrl']);
      // Azubi/Ausbilder darf Urlaub aktualisieren
      allow update: if request.auth != null
        && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['urlaub']);
      // Azubi darf Benachrichtigungen als gelesen markieren + Kurs-Entfernung bestätigen
      allow update: if request.auth != null
        && request.resource.data.diff(resource.data).affectedKeys().hasOnly(
          ['benachrichtigungen','aktiverKursId','pendingRemoval','pendingRemovalKurs','kursHistory']
        );
      // TN-Dokumente (Krankmeldungen, Bescheinigungen)
      match /dokumente/{fileDoc} {
        allow read: if request.auth != null;
        allow create, delete: if request.auth != null;
      }
    }

    // Mail-Collection (Firebase Trigger Email Extension)
    // Alle authentifizierten User dürfen Mails erstellen (für Benachrichtigungen an Ausbilder/AG)
    match /mail/{doc} {
      allow create: if request.auth != null;
      allow read, update, delete: if isAdmin();
    }
  }
}

Einstellungen
5▾
Verwaltung-MA – Name, E-Mail und Passwort vergeben. Account wird automatisch erstellt.
Ausbilder – Name, E-Mail, Passwort und Berufsgruppe. Login-Account wird automatisch erstellt.
Lehrgangsarten – Kürzel und Beschreibung anlegen. Erscheinen als Dropdown beim Kurs-Erstellen.
Berufsgruppen – Name, Emoji und Fachrichtungen. Werden in AzubiPro und ArbeitgeberPro als Dropdown angezeigt.
AzubiPro-Konten – Beim TN-Anlegen optional Startpasswort vergeben und Zugangskarte drucken.

Urlaubsplanung
4▾

Ausbilder-Urlaub – Im Ausbilder-Profil unter Einstellungen eintragen.
TN-Urlaub – Im TN-Profil eintragen. Azubis können ihren Urlaub auch selbst in AzubiPro pflegen.
Kurs-Builder – Ausbilder und TN mit Urlaub werden beim Erstellen automatisch blockiert.
Gut zu wissen
Arbeitgeber können sich selbständig registrieren und werden unter Betriebe angezeigt. Wenn ein AG einen Azubi anlegt, erscheint dieser automatisch in der TN-Liste. Als Admin siehst du im Dashboard wer wann welche App genutzt hat.
