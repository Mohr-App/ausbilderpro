rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isAdmin() {
      return request.auth != null
        && (get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin'
         || get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'verwaltung');
    }

    match /users/{uid} {
      allow read, write: if request.auth != null && request.auth.uid == uid;
      allow read, write: if isAdmin();
      match /{sub}/{subDoc} {
        allow read, write: if request.auth != null && request.auth.uid == uid;
        allow read, write: if isAdmin();
        match /{deep}/{deepDoc} {
          allow read, write: if request.auth != null && request.auth.uid == uid;
          allow read, write: if isAdmin();
        }
      }
    }

    match /shared/{docId} {
      allow read: if request.auth != null;
      allow write: if isAdmin();
      match /{sub}/{subDoc} {
        allow read: if request.auth != null;
        allow write: if isAdmin();
        // Standalone: Ausbilder darf Kurse erstellen
        allow create: if request.auth != null && docId == 'kurse' && sub == 'list';
        // Standalone: Ausbilder darf eigene Kurse löschen
        allow delete: if request.auth != null && docId == 'kurse' && sub == 'list'
          && resource.data.createdBy == request.auth.token.email;
        allow update: if request.auth != null
          && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['urlaub']);
        allow update: if request.auth != null
          && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['benachrichtigungen']);
        allow update: if request.auth != null
          && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['berufsschultage']);
        allow update: if request.auth != null
          && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['kurstage']);
        allow create: if request.auth != null && docId == 'urlaubLog';
        match /{deep}/{deepDoc} {
          allow read: if request.auth != null;
          allow write: if isAdmin();
          // Selbstregistrierung: Azubi darf sich in Kurs eintragen
          allow create: if request.auth != null && deep == 'teilnehmer';
          // Ausbilder darf TN aus Kurs löschen
          allow delete: if request.auth != null && deep == 'teilnehmer';
          allow update: if request.auth != null
            && request.resource.data.diff(resource.data).affectedKeys().hasOnly(
              ['status','attTotal','attPresent','attAbsentE','attAbsentU','attSick','attBerufsschule','attLate','attEarly','attPercent','attLastSync','noten_mitarbeit','noten_fuehrung','noten_praxis','noten_extra','werkzeugStatus','werkzeugFehlt','bemerkung','bemerkungVon','bescheinigungSignatur','bescheinigungDatum','bescheinigungVon']
            );
          allow update: if request.auth != null
            && request.resource.data.diff(resource.data).affectedKeys().hasOnly(
              ['bestaetigtAm','bestaetigtVon']
            );
          allow update: if request.auth != null
            && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['wunschSitzplatz']);
          allow create, delete: if request.auth != null && deep == 'dokumente';
          // Meldungen: Azubi darf erstellen, Ausbilder darf lesen/löschen
          allow create: if request.auth != null && deep == 'meldungen';
          allow delete: if request.auth != null && deep == 'meldungen';
          allow update: if request.auth != null && deep == 'meldungen'
            && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['gelesen']);
        }
      }
    }

    match /globalTN/{doc} {
      allow read: if request.auth != null;
      allow write: if isAdmin();
      allow delete: if request.auth != null;
      // Selbstregistrierung + manuelles Anlegen
      allow create: if request.auth != null;
      allow update: if request.auth != null
        && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['status', 'wiederholer']);
      allow update: if request.auth != null
        && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['photoUrl']);
      allow update: if request.auth != null
        && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['kontaktEmail','kontaktTelefon','wunschSitzplatz','dsgvoZustimmung']);
      allow update: if request.auth != null
        && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['urlaub']);
      allow update: if request.auth != null
        && request.resource.data.diff(resource.data).affectedKeys().hasOnly(
          ['benachrichtigungen','aktiverKursId','pendingRemoval','pendingRemovalKurs','kursHistory','wunschSitzplatz']
        );
      allow update: if request.auth != null
        && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['beruf']);
      match /dokumente/{fileDoc} {
        allow read: if request.auth != null;
        allow create, delete: if request.auth != null;
      }
      match /zeiterfassung/{dayDoc} {
        allow read, write: if request.auth != null;
      }
      match /urlaubsantraege/{antragDoc} {
        allow read, write: if request.auth != null;
      }
      match /baustellen/{bsDoc} {
        allow read, write: if request.auth != null;
      }
    }

    match /mail/{mailId} {
      allow create: if request.auth != null;
      allow read, update, delete: if false; // Nur Extension
    }
  }
}
